// 主函数
async function moveToPlayerNearby(bot) {
  // 获取最近的玩家实体
  const player = bot.nearestEntity(entity => entity.type === 'player');
  if (!player) {
    bot.chat('附近没有找到玩家！');
    return;
  }

  // 获取玩家位置
  const playerPos = player.position;
  bot.chat(`正在前往玩家位置附近: x=${playerPos.x}, y=${playerPos.y}, z=${playerPos.z}`);

  // 移动到玩家位置附近3个格子的范围内
  await bot.pathfinder.goto(new GoalNear(playerPos.x, playerPos.y, playerPos.z, 3));
  bot.chat('已到达玩家位置附近！');
}
